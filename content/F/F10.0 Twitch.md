```dataviewjs
dv.el("style", `
    .dataview table tbody tr {
        height: 10px; /* Adjust this value to make rows thinner */
    }
`);


// Safely get all pages in "content" folder
const pages = dv.pages('"content"')
    .array()
    .filter(p => p.area === "Twitch");

// Handle no results
if (pages.length === 0) {
    dv.paragraph("⚠️ No notes with area: Twitch found in folder 'content'.");
} else {
    dv.table(
        ["Work", "Description", "Completion", "Duration", "Skills", "Team", "Progress"],
        pages.map(p => {
            // ✅ Team checkboxes rendering
            const team = Array.isArray(p.team)
                ? p.team.map(member => {
                    if (typeof member === "object" && "checked" in member) {
                        return member.checked ? "✅" : "◻️";
                    } else {
                        return "◻️";
                    }
                }).join(" ")
                : "–";

            // ✅ Task progress
            const tasks = p.file.tasks || [];
            const completed = tasks.filter(t => t.completed).length;
            const total = tasks.length;
            const percent = total === 0 ? 0 : Math.round((completed / total) * 100);

            // ✅ Progress bar
            const barContainer = dv.el("div", "", {
                attr: {
                    style: `
                        width: 100%;
                        background-color: #ddd;
                        border-radius: 5px;
                        height: 10px;
                        position: relative;
                    `
                }
            });

            dv.el("div", "", {
                attr: {
                    style: `
                        width: ${percent}%;
                        background-color: #9146FF;
                        border-radius: 5px;
                        height: 10px;
                        margin-top: -10px;
                        position: relative;
                    `
                },
                parent: barContainer
            });

            return [
                dv.fileLink(p.file.name),
                p.description ?? "-",
                p.completed_on ?? "-",
                p.project_duration ?? "-",
                (p.skills ?? []).map(s => dv.fileLink(s)).join(", "),
                team,
                barContainer
            ];
        })
    );
}
```
